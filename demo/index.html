Builder Surface

TypeScript SDK
import { Sluice } from 'sluice/sdk';

const sluice = new Sluice({ url: '...' });

// 1. Check readiness
const status = await sluice.checkReadiness(peerId);

// 2. Prepare inbound liquidity
if (status.needsInbound) {
  const quote = await sluice.quote({ amount: "1" });
  await sluice.prepareInbound(peerId, quote);
}
HTTP API
# Request inbound channel quote
curl -X POST http://localhost:8787/v1/quote \
  -H "Content-Type: application/json" \
  -d '{"target_payment": "100000000"}'

# Prepare the channel
curl -X POST http://localhost:8787/v1/prepare \
  -H "Content-Type: application/json" \
  -d '{"peer_id": "...", "execute": true}'
